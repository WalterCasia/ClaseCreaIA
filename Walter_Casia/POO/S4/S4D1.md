# Modelo - vista - controlador (MVC)
## las 3 capas

* Modelo (model): es el cerebro, maneja los datos y la logica del negocio.
* Vista (view): es la cara, se encarga de mostrar los datos al usuario.
* Controlador (controller): es el intermediario, recibe las peticiones del usuario y las envia al modelo.

```python
# 1. Modelo (Datos y Lógica)
class CalculadoraModel:
    def sumar(self, a, b):
        return a + b

    def restar(self, a, b):
        return a - b

# 2. Vista (Presentación)
class CalculadoraView:
    def mostrar_resultado(self, resultado):
        print(f"El resultado es: {resultado}")

    def mostrar_error(self, mensaje):
        print(f"Error: {mensaje}")

# 3. Controlador (Orquestación)
class CalculadoraController:
    def __init__(self, model, view):
        self.model = model
        self.view = view

    def manejar_suma(self, num1, num2):
        try:
            resultado = self.model.sumar(num1, num2)
            self.view.mostrar_resultado(resultado)
        except Exception as e:
            self.view.mostrar_error(str(e))

# Flujo de ejecución
if __name__ == "__main__":
    # 1. Crear instancias
    modelo = CalculadoraModel()
    vista = CalculadoraView()
    controlador = CalculadoraController(modelo, vista)

    # 2. El usuario interactúa (simulado)
    print("--- Calculadora MVC ---")
    try:
        n1 = float(input("Ingrese el primer número: "))
        n2 = float(input("Ingrese el segundo número: "))
        operacion = input("Ingrese la operación (+ o -): ")

        # 3. El controlador decide qué hacer
        if operacion == "+":
            controlador.manejar_suma(n1, n2)
        elif operacion == "-":
            controlador.manejar_resta(n1, n2) # Asumiendo que creamos manejar_resta
        else:
            vista.mostrar_error("Operación no válida")

    except ValueError:
        vista.mostrar_error("Entrada inválida")
```
